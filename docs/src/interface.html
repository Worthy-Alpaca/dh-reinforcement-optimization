178        """Method to center the current window on screen.
179
180        Args:
181            master (tk.Toplevel): The current master.
182            width (int, optional): The current width. Defaults to 300.
183            height (int, optional): The current height. Defaults to 200.
184
185        Returns:
186            str: calculated screen coordinates for centering the window.
187        """
188        # get screen width and height
189        screen_width = master.winfo_screenwidth()
190        screen_height = master.winfo_screenheight()
191
192        # calculate position x and y coordinates
193        x = (screen_width / 2) - (width / 2)
194        y = (screen_height / 2) - (height / 2)
195        return "%dx%d+%d+%d" % (width, height, x, y)

334    def __optimize(self):
335        """Method to start the optimisation process.
336
337        Returns:
338            None: None
339        """
340        self.text.config(state=NORMAL)
341        self.text.delete(1.0, tk.END)
342
343        try:
344            startDate = self.calDate["start"]
345            endDate = self.calDate["end"]
346        except:
347            self.controller.error("Bitte setzten Sie eine Datumsspanne.")
348            return warning("Please set a date range.")
349
350        if self.optimizerData == None:
351            datapath = self.__openNew(
352                startPath=os.path.expanduser(os.path.normpath("~/Downloads")),
353                filetypes=[("excel spreadsheet", ".xlsx")],
354            )
355            if datapath == None:
356                return
357        else:
358            return
359
360        self.controller.wait(message="Datenset generierung...")
361
362        runmodel = RunModel(
363            dbpath=self.config.get("optimizer_backend", "dbpath"),
364            numSamples=self.config.getint("model_training", "trainingSamples"),
365            caching=self.config.getboolean("default", "useCache"),
366            disableProgress=self.config.getboolean("default", "progressBar"),
367            overwriteDevice="cpu",
368            refEngine=self.refEngine,
369        )
370        loader = KappaLoader(
371            os.path.normpath(datapath),
372            self.config.get("optimizer_backend", "dbpath"),
373            startDate,
374            endDate,
375        )
376        samples, sampleReqs, s_np = loader.getData()
377        if exists(self.basePath + os.path.normpath("/models")):
378            pathDir = self.basePath + os.path.normpath("/models")
379        else:
380            pathDir = self.resource_path("/bin/assets/models")
381        best_time = -float("inf")
382        best_value = any
383        best_solution = any
384        solutionIterator = self.config.getint("default", "solutionIterator")
385        for x in range(solutionIterator):
386            info(f"Iteration {x+1}/{solutionIterator}")
387            running_value, running_solution = runmodel.getBestOder(
388                sampleReqs=sampleReqs,
389                samples=samples,
390                debug=False,
391                numCarts=self.config.getint("default", "numCarts"),
392                modelFolder=pathDir,
393            )
394            running_time = runmodel.helper.calc_total_time(running_solution["solution"])
395            if running_time[0] > best_time:
396                best_time = running_time[0]
397                best_value = running_value
398                best_solution = running_solution
399        self.controller(
400            best_value,
401            best_solution,
402            s_np,
403            dbpath=self.config.get("optimizer_backend", "dbpath"),
404            calcGroups=self.config.getboolean("default", "calcgroups"),
405            overlapThreshhold=0.5,
406            refEngine=self.refEngine,
407            textSeparator=self.config.get("default", "textSeparator"),
408        )
409        return

412        """Method to train the modell."""
413        if self.__askQuestion(

415            "Möchten Sie wirklich ein neues Training starten?\nDies kann mehrere Stunden dauern.",
416            height=200,

424            runmodel = RunModel(
425                dbpath=self.config.get("optimizer_backend", "dbpath"),
426                numSamples=self.config.getint("model_training", "trainingSamples"),

430            EMBEDDING_DIMENSIONS = 28
431            EMBEDDING_ITERATIONS_T = 2
432            optim_args = {

434                "eps": 0.2832233236834357,
435            }
436            Q_Function, QNet, Adam, ExponentialLR = runmodel.init_model(
437                EMBEDDING_DIMENSIONS=EMBEDDING_DIMENSIONS,
438                EMBEDDING_ITERATIONS_T=EMBEDDING_ITERATIONS_T,
439                INIT_LR=0.0036528384837269156,
440                OPTIMIZER=torch.optim.Adam,
441                loss_func=torch.nn.HuberLoss,
442                optim_args=optim_args,
443            )
444            runmodel.fit(
445                Q_func=Q_Function,
446                Q_net=QNet,
447                optimizer=Adam,
448                lr_scheduler=ExponentialLR,
449                NR_EPISODES=self.config.getint("model_training", "trainingEpisodes"),

454                GAMMA=self.config.getfloat("model_training", "gamma"),
455                debug=True,
