172        EMBEDDING_ITERATIONS_T: int = 2,

177        LR_DECAY_RATE: float = 1.0 - 2e-5,
178        debug: bool = True,

180        """Initiate the model state loading. If no `fname` is given, a new model will be initialized.
181
182        Args:
183            fname (str, optional): Path to a saved state_dict. Defaults to None.
184            EMBEDDING_DIMENSIONS (int, optional): Embedding dimensions in the model. Defaults to 10.
185            EMBEDDING_ITERATIONS_T (int, optional): Embedding iterations in the model. Defaults to 2.
186            INIT_LR (float, optional): Initial learning rate. Changes over time. Defaults to 3e-3.
187            OPTIMIZER (FunctionType, optional): The Optimizer to use. Defaults to torch.optim.Adam.
188            optim_args (dict, optional): Optional keyword arguments for optimizer instance. Defaults to {}.
189            loss_func (FunctionType, optional): The loss function to use. Defaults to torch.nn.MSELoss.
190            LR_DECAY_RATE (float, optional): The decay rate of the initial learning rate. Defaults to 1.0-2e-5.
191            debug (bool, optional): If debug information should be displayed. Defaults to True.

193        Returns:
194            tuple: Q Function, Q Networ, optimizer, learning rate scheduler
195        """
196        self.embedding_dimensions = EMBEDDING_DIMENSIONS
197        self.embedding_iterations_t = EMBEDDING_ITERATIONS_T
198        Q_net = QNetModel(
199            device=self.device,
200            emb_dim=EMBEDDING_DIMENSIONS,
201            emb_it=EMBEDDING_ITERATIONS_T,
202        )

204        optimizer = OPTIMIZER(Q_net.parameters(), lr=INIT_LR, **optim_args)
205        lr_scheduler = torch.optim.lr_scheduler.ExponentialLR(
206            optimizer, gamma=LR_DECAY_RATE
207        )

210            f"{OPTIMIZER.__class__.__name__}_{EMBEDDING_DIMENSIONS}@{timestamp}"

219            Q_net.load_state_dict(checkpoint["model"])
220            optimizer.load_state_dict(checkpoint["optimizer"])
221            lr_scheduler.load_state_dict(checkpoint["lr_scheduler"])
222
223        Q_func = QFunction(
224            Q_net, optimizer, lr_scheduler, device=self.device, loss_fn=loss_func
225        )
226        return Q_func, Q_net, optimizer, lr_scheduler

268    def state2tens(self, state: NamedTuple) -> torch.tensor:
269        """Method to convert a given state into PyTorch tensor.
270
271        Args:
272            state (NamedTuple): The current state.
273
274        Returns:
275            torch.tensor: The created tensor.
276        """
277        solution = set(state.partial_solution)
278        sol_last_node = (
279            state.partial_solution[-1] if len(state.partial_solution) > 0 else -1
280        )
281        sol_first_node = (
282            state.partial_solution[0] if len(state.partial_solution) > 0 else -1
283        )
284        coords = state.coords
285        nr_nodes = coords.shape[0]
286
287        xv = [
288            [
289                (1 if i in solution else 0),
290                (1 if i == sol_first_node else 0),
291                (1 if i == sol_last_node else 0),
292                coords[i, 0],
293                coords[i, 1],
294                coords[i, 2],
295            ]
296            for i in range(nr_nodes)
297        ]
298
299        return torch.tensor(
300            xv, dtype=torch.float, requires_grad=False, device=self.device
301        )

321    def get_duplicates(self, iterable: list):
322        """Method to retrieve duplicates in given iterable.
323
324        Args:
325            iterable (list): Iterable to be searched for duplicates
326
327        Returns:
328            tuple: found duplicates, their occurance
329        """
330        proofingDict = {}
331        for el in iterable:
332            try:
333                proofingDict[el] += 1
334            except:
335                proofingDict[el] = 1
336        duplicates = []
337        occurances = []
338        for key in proofingDict.keys():
339            for i in range(proofingDict[key] - 1):
340                duplicates.append(key)
341                occurances.append(iterable.count(key))
342        return duplicates, occurances

344    def getData(self, samples: list = False, sampleReqsList: list = False):
345        """Method to generate a data sample from a given sample list.
346
347        Args:
348            samples (list, optional): The current sample. Defaults to False.
349            sampleReqsList (list, optional): The current sample requirements. Defaults to False.
350        """
351
352        def compare(p1, p2):
353            l2 = list(set(p1) & set(p2))
354            l1 = len(p1)
355            return l2
356
357        if samples:
358            x = compare(self.products, samples)
359            sampleSize = samples
360        else:
361            sampleSize = self.getRandomSample(self.numSamples)
362
363        if sampleReqsList:
364            sampleReqs = sampleReqsList
365        else:
366            sampleReqs = np.random.randint(1, 70, size=(len(sampleSize))).tolist()
367
368        currentDuplicates, occurances = self.get_duplicates(sampleSize)
369        for x in currentDuplicates:
370            lettersIndexes = [i for i in range(len(sampleSize)) if sampleSize[i] == x]
371            runningReqs = 0
372            for i in lettersIndexes:
373                runningReqs += sampleReqs[i]
374            lettersIndexes.reverse()
375            for i in lettersIndexes:
376                del sampleReqs[i]
377                del sampleSize[i]
378            lettersIndexes.reverse()
379            sampleReqs.insert(lettersIndexes[0], runningReqs)
380            sampleSize.insert(lettersIndexes[0], x)
381
382        def createCoords(sampleSize):
383
384            globalList = {}
385            currentDict = []
386            for i in sampleSize:
387                currentList = sampleSize.copy()
388                currentList.remove(i)
389                req = sampleReqs[sampleSize.index(i)]
390                simTime = (
391                    self.products[i]["time"] if self.products[i]["time"] != 0 else 0
392                )
393                if simTime > 1000:
394                    simTime = simTime / 100
395                currentDict.append(
396                    [
397                        math.log(float(self.products[i]["len"])) / 10
398                        if float(self.products[i]["len"]) != 0
399                        else 0,
400                        math.log(simTime) / 10,
401                        math.log(self.products[i]["score"]) / 10
402                        if self.products[i]["score"] != 0
403                        else 0,
404                        i,
405                        self.products[i]["comps"],
406                        float(len(self.products[i]["comps"])),
407                        req,
408                    ]
409                )
410
411                globalList[i] = currentDict
412            return globalList
413
414        globalList = createCoords(sampleSize)
415        npArray = []
416
417        if len(globalList) < self.numSamples and self.training:
418            supplement = self.getRandomSample(
419                self.numSamples - len(globalList), sampleSize
420            )
421            supplement = supplement + sampleSize
422            globalList = createCoords(supplement)
423
424        product, coords = random.choice(list(globalList.items()))
425
426        for i in coords:
427            npArray.append(i)
428
429        coords = np.asarray(coords, dtype=object)
430        W_np = distance_matrix(
431            coords[:, :3].astype(np.float32), coords[:, :3].astype(np.float32)
432        )
433        return coords, W_np, product

479    def fit(
480        self,
481        Q_func: FunctionType,
482        Q_net: nn.Module,
483        optimizer: FunctionType,
484        lr_scheduler: FunctionType,
485        NR_EPISODES: int,
486        MIN_EPSILON: float,
487        EPSILON_DECAY_RATE: float,
488        N_STEP_QL: int,
489        BATCH_SIZE: int,
490        GAMMA: float,
491        trial: optuna.trial.Trial = None,
492        debug: bool = False,
493    ):
494        """Train the current model.
495
496        Args:
497            Q_func (FunctionType): The current Q_function
498            Q_net (nn.Module): The current model.
499            optimizer (FunctionType): The current optimizer function.
500            lr_scheduler (FunctionType): The current learning rate scheduler.
501            NR_EPISODES (int): The number of training iterations.
502            MIN_EPSILON (float): The minimum epsilon value.
503            EPSILON_DECAY_RATE (float): The current epsilon decay rate.
504            N_STEP_QL (int): Placeholder
505            BATCH_SIZE (int): The current batch size.
506            GAMMA (float): The current gamma value.
507            trial (optuna.trial.Trial): The current trial in tuning. Defaults to None.
508            debug (bool): If debug information should be displayed. Defaults to None.
509        Returns:
510            float: The median loss value for the training.
511        """
512        self.training = True
513        self.losses = []
514        self.lrs = []
515        self.path_lengths = []
516        self.rewards = []
517        current_min_med_length = float("inf")
518        Q_net = Q_net.float()
519        # BATCH_SIZE = self.numSamples
520        productDataset = ProductDataset(*self.generateData())
521        productDataloader = ProductDataloader(
522            productDataset,
523            self.numSamples,
524            shuffle=True,
525            num_workers=4,
526            drop_last=True,
527            persistent_workers=True,
528            pin_memory=True,
529        )
530        step = -1
531        for episode in tqdm(range(NR_EPISODES), disable=self.disableProgress):
532            coords, W_np, components = next(iter(productDataloader))
533
534            coords, W, components = (
535                coords.to(self.device, non_blocking=True).float(),
536                torch.tensor(
537                    W_np,
538                    device=self.device,
539                    dtype=torch.float,
540                    requires_grad=False,
541                ),
542                components.to(self.device, non_blocking=True),
543            )
544            self.helper = UtilFunctions(components)
545
546            # current partial solution - a list of node index
547            solution = [random.randint(0, coords.shape[0] - 1)]
548
549            # current state (tuple and tensor)
550            current_state = self.State(partial_solution=solution, W=W, coords=coords)
551            current_state_tsr = self.state2tens(current_state)
552
553            # Keep track of some variables for insertion in replay memory:
554            states = [current_state]
555            states_tsrs = [
556                current_state_tsr
557            ]  # we also keep the state tensors here (for efficiency)
558            rewards = []
559            actions = []
560
561            # current value of epsilon
562            epsilon = max(MIN_EPSILON, (1 - EPSILON_DECAY_RATE) ** episode)
563
564            nr_explores = 0
565            t = -1
566            while not self.helper.is_state_final(current_state):
567                t += 1  # time step of this episode
568                step += 1
569                if epsilon >= random.random():
570                    # explore
571                    next_node = self.helper.get_next_neighbor_random(current_state)
572                    nr_explores += 1
573                else:
574                    # exploit
575                    next_node, _ = Q_func.get_best_action(
576                        current_state_tsr, current_state
577                    )
578                del current_state_tsr, current_state
579                next_solution = solution + [next_node]
580
581                # reward observed for taking this step
582                reward = -(
583                    self.helper.total_distance(next_solution, W)[0]
584                    - self.helper.total_distance(solution, W)[0]
585                )
586                self.rewards.append(self.helper.total_distance(next_solution, W)[0])
587                next_state = self.State(
588                    partial_solution=next_solution,
589                    W=W,
590                    coords=coords,
591                )
592                next_state_tsr = self.state2tens(next_state)
593
594                # store rewards and states obtained along this episode:
595                states.append(next_state)
596                states_tsrs.append(next_state_tsr)
597                rewards.append(reward)
598                actions.append(next_node)
599
600                # store our experience in memory, using n-step Q-learning:
601                if len(solution) >= N_STEP_QL:
602                    self.memory.remember(
603                        self.Experience(
604                            state=states[-N_STEP_QL],
605                            state_tsr=states_tsrs[-N_STEP_QL],
606                            action=actions[-N_STEP_QL],
607                            reward=sum(rewards[-N_STEP_QL:]),
608                            next_state=next_state,
609                            next_state_tsr=next_state_tsr,
610                        )
611                    )
612
613                if self.helper.is_state_final(next_state):
614                    for n in range(1, N_STEP_QL):
615                        self.memory.remember(
616                            self.Experience(
617                                state=states[-n],
618                                state_tsr=states_tsrs[-n],
619                                action=actions[-n],
620                                reward=sum(rewards[-n:]),
621                                next_state=next_state,
622                                next_state_tsr=next_state_tsr,
623                            )
624                        )
625
626                # update state and current solution
627                current_state = next_state
628                current_state_tsr = next_state_tsr
629                solution = next_solution
630
631                # take a gradient step
632                loss = None
633                if len(self.memory) >= BATCH_SIZE and len(self.memory) >= 200:
634                    experiences = self.memory.sample_batch(BATCH_SIZE)
635
636                    batch_states_tsrs = [e.state_tsr for e in experiences]
637                    batch_Ws = [e.state.W for e in experiences]
638                    batch_actions = [e.action for e in experiences]
639                    batch_targets = []
640
641                    for i, experience in enumerate(experiences):
642                        target = experience.reward
643                        if not self.helper.is_state_final(experience.next_state):
644                            _, best_reward = Q_func.get_best_action(
645                                experience.next_state_tsr, experience.next_state
646                            )
647                            target += GAMMA * best_reward
648                        batch_targets.append(target)
649                        del target
650
651                    loss = Q_func.batch_update(
652                        batch_states_tsrs, batch_Ws, batch_actions, batch_targets
653                    )
654                    self.losses.append(loss)
655                    self.lrs.append(Q_func.optimizer.param_groups[0]["lr"])
